\section*{Easy (10 Questions)}
\begin{enumerate}[label=E\arabic*., leftmargin=*]
  \item \textbf{AI definition:} Define AI in 2--3 lines and give one real example system.

  \item \textbf{AI techniques:} Name three broad AI technique categories and give one example for each.

  \item \textbf{Success criteria:} Give three success criteria (beyond accuracy) for an AI-based \textbf{loan approval support} system.

  \item \textbf{Agent basics:} Define an \textbf{agent}. Give one example and list its sensors and actuators.

  \item \textbf{PEAS:} Write a PEAS description for a \textbf{spam filter}.

  \item \textbf{PL translation:} Let $P$ = ``It is raining'' and $Q$ = ``Ground is wet''.
    Translate: ``If it is raining then the ground is wet, and it is raining.'' Write the formula and state what it entails.

  \item \textbf{Truth table (core):} Write the truth table for $P \rightarrow Q$ and state when it is false.

  \item \textbf{Validity/satisfiability:} Define \textbf{valid} and \textbf{satisfiable} (one example each).

  \item \textbf{Entailment vs implication:} Explain (6--10 lines) the difference between $KB \models \alpha$ and $KB \rightarrow \alpha$.

  \item \textbf{FOL basics:} In $Teaches(Prof, Course)$ identify: predicate symbol, constants, and variables (if any).
\end{enumerate}

\section*{Medium (10 Questions)}
\begin{enumerate}[label=M\arabic*., leftmargin=*]
  \item \textbf{Environment properties:} Classify the environment of \textbf{self-driving in traffic} using at least five properties
    (observable, deterministic, episodic, static, discrete, etc.). Justify briefly.

  \item \textbf{Learning agent:} List the four main components of a learning agent and explain the role of the critic (6--8 lines).

  \item \textbf{Agent selection:} Choose an agent type (simple reflex / model-based / goal-based / utility-based / learning)
    for a \textbf{campus navigation assistant}. Justify in 8--12 lines.

  \item \textbf{Limits and mitigation:} List four deployment-time limitations of AI systems and one mitigation for each.

  \item \textbf{Model checking:} Let $KB=\{P \rightarrow Q,\ Q \rightarrow R,\ P\}$.
    Check whether $KB \models R$ using a truth table or model enumeration. Show the key steps.

  \item \textbf{CNF conversion:} Convert $\lnot(P \rightarrow Q)$ to CNF (show steps).

  \item \textbf{CNF (pipeline):} Convert $(P \rightarrow Q) \land (Q \rightarrow R)$ to CNF.

  \item \textbf{Resolution (one proof):} Using resolution refutation, prove:
    \[
      \{P \rightarrow Q,\ Q \rightarrow R,\ P\} \models R.
    \]
    (Show clauses and the main resolution steps.)

  \item \textbf{English $\rightarrow$ FOL:} Domain: students and courses. Translate into FOL:
    \begin{enumerate}[label=(\alph*), leftmargin=*]
      \item Every student studies at least one course.
      \item Some course is studied by every student.
    \end{enumerate}

  \item \textbf{Unification (MGU):} Find an MGU if it exists:
    \[
      Knows(John, x)\ \text{and}\ Knows(y, Bill).
    \]
    Show the substitution.
\end{enumerate}

\section*{Hard (10 Questions)}
\begin{enumerate}[label=H\arabic*., leftmargin=*]
  \item \textbf{Knowledge-based agent:} Describe a knowledge-based agent for a \textbf{smart home security} scenario.
    Provide: PEAS, 6--10 propositional symbols, at least 6 rules/facts, and one query the agent should answer using entailment.

  \item \textbf{Integrated reasoning:} A campus assistant must decide whether to send an alert.
    Create a propositional KB (at least 6 sentences) that models: fire alarm, smoke detected, system fault, and manual override.
    Ask two queries and answer using model checking or resolution.

  \item \textbf{Resolution proof (chain):} Prove using resolution refutation:
    \[
      \{A \rightarrow B,\ B \rightarrow C,\ C \rightarrow D,\ A\} \models D.
    \]

  \item \textbf{Non-trivial CNF:} Convert to CNF:
    \[
      \lnot(P \leftrightarrow Q) \rightarrow (R \lor P).
    \]
    Show key steps, including handling $\leftrightarrow$.

  \item \textbf{Clause form (FOL):} Convert to clause form (show Skolemization):
    \[
      \forall x\ (Student(x) \rightarrow \exists y\ Mentors(y,x)).
    \]

  \item \textbf{FOL resolution (mini proof):} From
    \[
      \forall x\ (Human(x) \rightarrow Mortal(x)),\quad Human(Socrates)
    \]
    prove $Mortal(Socrates)$ using resolution (convert to clauses and refute $\lnot Mortal(Socrates)$).

  \item \textbf{Quantifier scope ambiguity:} Write two FOL translations for:
    ``Every student likes a course.''
    Explain the meaning difference (10--12 lines).

  \item \textbf{Occurs-check:} Explain (6--10 lines) why unifying $x$ with $f(x)$ should fail.
    What could go wrong in inference if we allow it?

  \item \textbf{Countermodel:} Provide a small KB and show a countermodel to prove $KB \not\models \alpha$.
    (Write the interpretation/assignment clearly.)

  \item \textbf{Reflection (logic in AI):} In 12--16 lines, explain how entailment, satisfiability, CNF, and resolution
    fit together to support automated reasoning in AI systems (knowledge bases, rule engines, verification, etc.).
\end{enumerate}
