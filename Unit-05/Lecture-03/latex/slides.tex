\section{Domains}
\label{sec:domains}

\begin{frame}{Learning Outcomes}
  By the end of this lecture, you should be able to:
  \begin{itemize}[<+->]
    \item List ML use-cases in education, insurance, retail/supply chain, transport, and energy.
    \item Explain why domain constraints change metric choice and deployment strategy.
    \item Apply a simple ``what could go wrong'' checklist to an AI system.
    \item Answer key revision questions across Units I--V.
  \end{itemize}
\end{frame}

\begin{frame}{Abbreviations \& Keywords}
  \begin{itemize}[<+->]
    \item KPI: key performance indicator \hfill ETA: estimated time of arrival
    \item Drift: patterns change over time \hfill Feedback loop: model affects data
    \item SLA: service-level agreement (latency/uptime constraints)
    \item Objective mismatch: wrong metric for the real goal
  \end{itemize}
\end{frame}

\begin{frame}{Education}
  \begin{itemize}[<+->]
    \item personalization / recommendations
    \item automated feedback and assessment support
    \item dropout risk prediction
  \end{itemize}
  Key concerns: fairness, privacy of student data, transparency.
\end{frame}

\begin{frame}{Insurance}
  \begin{itemize}[<+->]
    \item risk modeling and pricing support
    \item claim triage and fraud detection
    \item customer support automation
  \end{itemize}
  Key concerns: explainability, regulatory expectations, fairness.
\end{frame}

\begin{frame}{Retail \& Supply Chain}
  \begin{itemize}[<+->]
    \item demand forecasting
    \item recommendation systems
    \item inventory optimization
  \end{itemize}
  Key concerns: drift (seasonality), feedback loops, data quality.
\end{frame}

\begin{frame}{Transportation \& Logistics}
  \begin{itemize}[<+->]
    \item route optimization
    \item ETA prediction
    \item predictive maintenance
  \end{itemize}
  Key concerns: safety, real-time constraints, sensor reliability.
\end{frame}

\begin{frame}{Energy \& Utilities}
  \begin{itemize}[<+->]
    \item load forecasting
    \item fault detection
    \item demand-response optimization
  \end{itemize}
  Key concerns: reliability, rare-event detection, monitoring.
\end{frame}

\section{Checklist}
\label{sec:check}

\begin{frame}{What Could Go Wrong? (Checklist)}
  \begin{enumerate}[<+->]
    \item Wrong objective/metric for the real goal
    \item Data leakage or non-representative data
    \item Bias/fairness issues across groups
    \item Drift after deployment (no monitoring)
    \item Security/privacy risks and misuse
  \end{enumerate}
\end{frame}

\section{Revision}
\label{sec:rev}

\begin{frame}{Revision Questions (Sample)}
  \begin{enumerate}
    \item What is entailment and how is it tested using refutation?
    \item Why do we need train/validation/test splits?
    \item Compare regression vs classification and their metrics.
    \item What does PCA optimize and why standardize before PCA?
    \item Give two examples of drift and how to monitor it.
  \end{enumerate}
\end{frame}
